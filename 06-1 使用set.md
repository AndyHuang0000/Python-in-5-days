## 2.範例
```
#-----------------------------
# 建立存放各線站名的set物件
#-----------------------------
red_stations=set()
blue_stations=set()
green_stations=set()

#---------------------------
# 將紅線站名加入set中
#---------------------------
file = open('mrt.txt', 'r')
for data in file.readlines():
    #處理單行資料
    data = data.strip()
    items = data.split(' ')
    
    #將各線站名加入各線的set中
    no=items[1]
    
    if no[0:2]=='RE':
        red_stations.add(items[2])
        
    if no[0:2]=='BL':
        blue_stations.add(items[2])        

    if no[0:2]=='GR':
        green_stations.add(items[2])    

#-----------------------------
# 印出各線的set內容
#-----------------------------
print(red_stations)
print('-'*30)

print(blue_stations)
print('-'*30)

print(green_stations)
print('-'*30)

#-------------
# 關檔 
#-------------
file.close()
```
