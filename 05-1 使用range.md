## 2.範例
```
#-----------------------
# 建立一個list物件
#-----------------------
stations=[]

#-----------------------------------------------
# 依序讀入檔案內容, 產生tuple物件, 加入list中.
#-----------------------------------------------
file = open('mrt.txt', 'r')
for data in file.readlines():
    data = data.strip()
    items = data.split(' ')
    
    #建立一個tuple物件
    station=(items[0], items[1], items[2])
    
    #將tuple物件加入list中
    stations.append(station)
    
#-----------------------------
# 用index依序取出list內容
#-----------------------------
for k in range(len(stations)):
    print(stations[k])

#-------------
# 關檔 
#-------------
file.close()
```
